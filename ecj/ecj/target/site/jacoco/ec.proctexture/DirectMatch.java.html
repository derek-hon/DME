package ec.proctexture;


import java.awt.Color;
import java.awt.image.*;

// import ec.proctexture.ImageToArrayStrategy;
import ec.proctexture.PSDExtensions.ImageToArrayStrategy;

public class DirectMatch {
    public BufferedImage img = null;
    public double[][] data = null;

    public DirectMatch(BufferedImage img) {
        this.img = img;
        data = PSDExtensions.ImageToDouble2D(this.img, null);
    }

    public double luminosityDistance(DirectMatch match) {
        double total = 0.0;
        for (int y = 0 ; y < this.img.getHeight() ; y ++) {
            for (int x = 0 ; x < this.img.getWidth() ; x ++) {
                total += Math.abs(this.data[x][y] - match.data[x][y]);
            }
        }
        return total;
    }
}
