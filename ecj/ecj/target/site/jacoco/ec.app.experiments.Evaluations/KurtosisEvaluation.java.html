package ec.app.experiments.Evaluations;

import java.awt.image.BufferedImage;

import ec.EvolutionState;
import ec.distributedME.*;
import ec.util.Parameter;


public class KurtosisEvaluation extends BehaviourEvaluation {

    int ImageSize;
    int[] values;

    public void setup(int[] values) {
        this.values = values;
    }
    
    public void setup(BufferedImage img) {
        // TODO Auto-generated method stub
        
    }

    @Override
    public void setup(int[][] pixels) {
        
    }

    @Override
    public double[] individualBehaviourCalculation(EvolutionState state, Elite ind) {        
        return new double[]{kurtosis()};
    }

    public double kurtosis() {
        double kurtosis = 0;

        for (int value : values)
            kurtosis = Math.pow(value - mean(), 4);

        return kurtosis / (values.length * Math.pow(standardDeviation(), 4));
    }

    public double mean() {
        double mean = 0;

        for (int value : values)
            mean += value;

        return mean/values.length;
    }

    public double standardDeviation() {
        double standardDeviation = 0;

        for (int value : values)
            standardDeviation += Math.pow(value - mean(), 2);

        return Math.sqrt(standardDeviation/values.length);
    }
}
