package ec.app.experiments.Evaluations;

import ec.EvolutionState;
import java.util.*;
import java.awt.image.BufferedImage;
import ec.util.Parameter;
import ec.distributedME.*;

/*
 * BehaviourEvaluation.java
 *
 * Created: 26 May, 2020
 * By: Derek Hon
 */

@SuppressWarnings("serial")
public class MeanRedMeanBlueEvaluation extends BehaviourEvaluation {
    /**
     * This method is to calculate the behaviours for the individual then returns
     * these values to be used.
     *
     * @param ind Individual to calculate behaviours for
     * @return double[] behaviour values
     */

     int ImageSize;
     int[] values;
     BufferedImage img;
     EvaluationFunctions functions;

     public void setup(int[] values) {
        this.values = values;
    }

    @Override
    public void setup(int[][] pixels) {
        functions = new EvaluationFunctions();
    }
    
    public void setup(BufferedImage img) {
        this.img = img;
    }

    public double[] individualBehaviourCalculation(EvolutionState state, Elite ind) {
        int[][] pixels = ind.getPixels();
        return new double[] { functions.averageRed(pixels), functions.averageBlue(pixels) };
    } // individualBehaviourCalculation

} // BehaviourEvaluation
