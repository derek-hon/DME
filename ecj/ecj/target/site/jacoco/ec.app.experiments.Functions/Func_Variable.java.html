package ec.app.experiments.Functions;

import static ec.app.experiments.GeneralandFitness.CoordinateVariables.*;

import ec.EvolutionState;
import ec.Problem;
import ec.app.experiments.GeneralandFitness.MultiData;
import ec.app.experiments.GeneralandFitness.TextureProblemForm;
import ec.app.experiments.GeneralandFitness.VanillaTextureProblemForm;
import ec.gp.ADFStack;
import ec.gp.GPData;
import ec.gp.GPIndividual;
import ec.gp.GPNode;
import ec.gp.SimplifyableGPNode;
import ec.util.Parameter;

@SuppressWarnings("serial")
public abstract class Func_Variable extends GPNode implements SimplifyableGPNode {
	@Override
	public int expectedChildren() {
		return 0;
	}

	@Override
	public boolean IsVolatile() {
		return true;
	}

	@Override
	public GPNode GetSimplifiedReplacement(EvolutionState state) {
		return this;
	}

	public static class Pos_X extends Func_Variable
	{
	    public String toString() { return "X"; }
	    public void eval(
	    	final EvolutionState state,
	        final int            thread,
	        final GPData         input,
	        final ADFStack       stack,
	        final GPIndividual   individual,
	        final Problem        problem
	    )
	    {
	        MultiData rd = ((MultiData)(input));
	        if (state.parameters.getBoolean(new Parameter("vanilla"), null, false))
				rd.SetTo(((VanillaTextureProblemForm) problem).Get_Current_Pos()[X.ordinal()]);
			else
				rd.SetTo(((TextureProblemForm) problem).Get_Current_Pos()[X.ordinal()]);
	    }
	}
	
	public static class Pos_Y extends Func_Variable
	{
	    public String toString() { return "Y"; }
	    public void eval(
	    	final EvolutionState state,
	        final int            thread,
	        final GPData         input,
	        final ADFStack       stack,
	        final GPIndividual   individual,
	        final Problem        problem
	    )
	    {
	        MultiData rd = ((MultiData)(input));
	        if (state.parameters.getBoolean(new Parameter("vanilla"), null, false))
				rd.SetTo(((VanillaTextureProblemForm) problem).Get_Current_Pos()[Y.ordinal()]);
			else
				rd.SetTo(((TextureProblemForm) problem).Get_Current_Pos()[Y.ordinal()]);
	    }
	}
}
